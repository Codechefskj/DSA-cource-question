class Solution {
public:
    int HelpershipWithinDays(vector<int>& weights, int cap) {
        int currday = 1 , currweight = 0 ;
        for(int i = 0 ; i< weights.size() ; i++){
            if(weights[i]<= cap-currweight){
                currweight += weights[i];
            }else{
                currday++;
                currweight = 0 ; 
                currweight = weights[i] ;
            }
        } return currday;
    }
    int shipWithinDays(vector<int>& weights, int days) {
        int st = INT_MIN;
        int ed =  0 ;
        for(int i = 0 ; i<weights.size() ; i++){
            st = max(st ,weights[i]);
            ed += weights[i];
        }
        int ans = -1 ;
        while(st<=ed){
        int mid = st+(ed-st)/2;
        int requiredDays = HelpershipWithinDays(weights, mid);
        if(requiredDays <= days){
            ans = mid;
            ed = mid -1 ;
        }else{
            st = mid+1;
        }
        }
        return ans ; 
    }
};

 class Solution {
  public:
  bool completemintime(const vector<int>& arr, int k, int required_time){
      int temp = 0, painter = 1;
      for(int i = 0; i < arr.size(); i++){
          if(arr[i] > required_time) return false;

          if(temp + arr[i] <= required_time){
              temp += arr[i];
          } else {
              painter++;
              temp = arr[i];
          }
      }
      return painter <= k;
  }

  int minTime(vector<int>& arr, int k) {
      int st = *max_element(arr.begin(), arr.end());
      int ed = accumulate(arr.begin(), arr.end(), 0);
      int ans = ed;

      while(st <= ed){
          int mid = st + (ed - st) / 2;
          if(completemintime(arr, k, mid)){
              ans = mid;
              ed = mid - 1;
          } else {
              st = mid + 1;
          }
      }
      return ans;
  }
};

Q) leetcode 1658. Minimum Operations to Reduce X to Zero?
ans)

class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int n = nums.size();

        // Build suffix sums: suffix[i] = sum of last i elements
        vector<int> suffix(n + 1, 0);
        for (int i = 0; i < n; i++) {
            suffix[i + 1] = suffix[i] + nums[n - 1 - i];
        }

        // Store suffix sum values in array: suffix_val[i] = sum of last i elements
        // Now try all prefix sums and use binary search on suffix to find (x - prefix)
        int prefix = 0;
        int ans = INT_MAX;

        // Traverse prefix from 0 to n
        for (int i = 0; i <= n; i++) {
            if (i > 0) prefix += nums[i - 1];
            if (prefix > x) break;

            // Find if x - prefix exists in suffix[]
            // We need to find j such that suffix[j] == (x - prefix) and i + j <= n
            int rem = x - prefix;

            // Binary search on suffix[] from 0 to n - i (no overlap)
            int l = 0, r = n - i;
            while (l <= r) {
                int mid = (l + r) / 2;
                if (suffix[mid] == rem) {
                    ans = min(ans, i + mid);
                    break;
                } else if (suffix[mid] < rem) {
                    l = mid + 1;
                } else {
                    r = mid - 1;
                }
            }
        }

        return (ans == INT_MAX) ? -1 : ans;
    }
};

#include <iostream>

    for (int i = 0; i < n; ++i) cin >> a[i];
    for (int i = 0; i < n; ++i) cin >> b[i];

    for (int i = 0; i < n; ++i) {
        diff[i] = a[i] - b[i];
    }

    sort(diff.begin(), diff.end());

    long long count = 0;
    for (int i = 0; i < n; ++i) {
        int pos = upper_bound(diff.begin(), diff.end(), -diff[i]) - diff.begin(); // this is noe form of using binary search 
        // We only consider j > i
        count += n - max(pos, i + 1);
    }
