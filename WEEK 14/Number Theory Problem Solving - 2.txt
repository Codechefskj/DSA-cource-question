int gcd(int a ,int b){
    if(min(a,b)==0){
        return max(a,b);
    }
    int maxi = max(a,b);
    int mini = min(a,b);
    return gcd(mini , maxi%mini);
}

(a*a)%m = ((a%m) * (a%m))%m
(a*a*a*a*a*a)%m = ((a%m)(a%m)(a%m)(a%m)(a%m))%m
(a^b)%m = ((a%m)^b)%m
// in this question we cannot calculate directly then we used write like this a^1000^2 * a^1000^1 * a^100^5 * a^10^3 * a^1^4 and same thing we code in cpp

class Solution {
public:
    int mod = 1337;
    int powerfunction(int a, int b) {
        a %= mod;
        if (b == 0) return 1;
        if (b == 1) return a;

        long long half = powerfunction(a, b / 2);
        long long ans = (half * half) % mod;

        if (b % 2 == 1) ans = (ans * a) % mod;

        return ans;
    }

    int superPow(int a, vector<int>& b) {
        if (b.empty()) return 1;

        int last = b.back();
        b.pop_back();

        int part1 = powerfunction(a, last);
        int part2 = powerfunction(superPow(a, b), 10);

        return (part1 * part2) % mod;
    }
};
