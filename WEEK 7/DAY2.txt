class NumArray {
public:
    vector<int> prefixsum;
    NumArray(vector<int>& nums) {
        prefixsum.resize(nums.size());
        prefixsum[0] = nums[0];
        for(int i = 1 ; i<nums.size() ; i++){
            prefixsum[i] = prefixsum[i-1]+nums[i];
        }
    }
    
    int sumRange(int left, int right) {
        if(left==0){ return prefixsum[right];}
        return prefixsum[right]-prefixsum[left-1];
    }
};

Q) " leetcode 2559 " count vowels string in the range ?
ans) 
class Solution {
public:
    vector<int> vowelStrings(vector<string>& words, vector<vector<int>>& queries) {
        int n = words.size();
        vector<int> vowels(n, 0);

        for (int i = 0; i < n; i++) {
            char first = words[i][0];
            char last = words[i].back(); 

            if ((first == 'a' || first == 'e' || first == 'i' || first == 'o' || first == 'u') &&
                (last == 'a' || last == 'e' || last == 'i' || last == 'o' || last == 'u')) {
                vowels[i] = 1;
            }
        }

        vector<int> prefixsum(n, 0);
        prefixsum[0] = vowels[0];
        for (int i = 1; i < n; i++) {
            prefixsum[i] = prefixsum[i - 1] + vowels[i];
        }

        int queriesize = queries.size();
        vector<int> finalans(queriesize, 0);
        for (int k = 0; k < queriesize; k++) {
            int firstrange = queries[k][0];
            int lastrange = queries[k][1];
            finalans[k] = prefixsum[lastrange] - (firstrange > 0 ? prefixsum[firstrange - 1] : 0);
        }

        return finalans;
    }
};
