Q) leetcode 31 next permutation ?
ans) 

In this question we check left hand side is sorted or not when no are break the sorted the we swap the no which is just greater than those no and then reverse the left over no 

class Solution {
    public void nextPermutation(int[] nums) {
        int i = nums.length - 2; 
        while (i >= 0 && nums[i] >= nums[i + 1]) {
            i--;
        }

        if (i >= 0) {
            int j = nums.length - 1;
            while (nums[j] <= nums[i]) {
                j--;
            }
            swap(nums, i, j);
        }
        reverse(nums, i + 1);
        }

    private void reverse(int[] nums, int start) {
        int i = start, j = nums.length - 1;
        while (i < j) {
            swap(nums, i, j);
            i++;
            j--;
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}

Q) gfg- longest substring "https://www.geeksforgeeks.org/problems/length-of-the-longest-substring3036/1"
ans) 

int lengthOfLongestSubstring(string &s) {
    int n = s.size();
    vector<int> repeatedstring(26, -1); // To store the last seen index of each character
    int maxLength = 0, start = 0; 

    for (int end = 0; end < n; end++) {
        int index = s[end] - 'a'; // Convert character to index (0 to 25)

        if (repeatedstring[index] != -1 && repeatedstring[index] >= start) {
            start = repeatedstring[index] + 1; // Move start to avoid duplicate
        }

        repeatedstring[index] = end; // Update last seen index
        maxLength = max(maxLength, end - start + 1);
    }

    return maxLength;
}

Q) leetcode 908 smallest range 1 ?
ans) 

New Difference=(maxi−k)−(mini+k) = maxi−mini−2∗𝑘

class Solution {
    public int smallestRangeI(int[] nums, int k) {
        int n = nums.length;
        int maxi = Integer.MIN_VALUE;
        int mini = Integer.MAX_VALUE;
        for(int i = 0 ; i<n ; i++){
            maxi = Math.max(maxi, nums[i]);
            mini = Math.min(mini , nums[i]);
        }
        return Math.max(0, maxi-mini - 2*k);
    }
}
