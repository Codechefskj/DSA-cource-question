long long addundermodulo(long a , long b ,long m){
  if(b==0) return 0;
  long halfans = addundermodulo(a,b/2,m);
  if(b%2==0){
    return (halfans+halfans)%m;
  }
  else{
    return ((halfans +halfans)%m + a )%m

Q) codeforces D. Divide and Equalize "https://codeforces.com/problemset/problem/1881/D"?
ans)
// in this question we calculate the total no of frequency of the elements and see we can equally divide the prime factor 
#include <iostream>
#include <vector>
#include <unordered_map>
using namespace std;

unordered_map<int,int> factors(const vector<int>& temp){
    unordered_map<int,int> ans;
    for(int x : temp){
        for(int j = 2; j*j <= x; j++){
            while(x % j == 0){
                ans[j]++;
                x /= j;
            }
        }
        if(x > 1) ans[x]++; 
    }
    return ans;
}

int main(){
    int t;
    cin >> t;
    while(t--){
        int n;
        cin >> n;
        vector<int> no(n);
        for(int i=0; i<n; i++) cin >> no[i];

        unordered_map<int,int> isfactor = factors(no);

        bool isvalid = true;
        for(auto m : isfactor){
            if(m.second % n != 0){ 
                isvalid = false;
                break; 
            }
        }
        if(isvalid) cout << "YES\n";
        else cout << "NO\n";
    }
}

Q) leetcode 2470. Number of Subarrays With LCM Equal to K?
ans) 
// we used LCM(a,b)= a*b/gcd(a,b) for every elements into subarray 

class Solution {
public:
    int gcd(int a , int b){
        if(min(a,b)==0){
            return max(a,b);
        }
        int mini = min(a,b);
        int maxi = max(a,b);
        return gcd(mini, maxi % mini);
    }

    int subarrayLCM(vector<int>& nums, int k) {
        int ans = 0; 
        for(int left = 0; left < nums.size(); left++){
            long long val = nums[left];  
            for(int right = left; right < nums.size(); right++){
                val = (val * nums[right]) / gcd(val, nums[right]);  
                if(val == k) ans++;        
                else if(val > k) break;   
            }
        }
        return ans;     
    }
};

class Solution {
public:
    int mod = 1000000007;

    pair<long long,long long> extendedeuclid(long long a , long long b){
        if(b == 0) return {1, 0};
        auto partial = extendedeuclid(b, a % b);
        long long x = partial.second;
        long long y = partial.first - (a / b) * partial.second;
        return {x, y};
    }

    long long modinverse(long long a , long long m){
        if(__gcd(a,m) != 1) return -1;
        long long x = extendedeuclid(a,m).first;
        return (x % m + m) % m;
    }

    long long factorial(long long d){
        long long ans = 1;
        for(long long i = 1 ; i <= d ; i++){
            ans = (ans * i) % mod;
        }
        return ans;
    }

    long long countpermulation(string s){
        unordered_map<char,int> m;
        for(char c : s) m[c]++;

        long long num = factorial(s.size());
        long long den = 1;
        for(auto &p : m){
            den = (den * factorial(p.second)) % mod;
        }
        long long inv = modinverse(den, mod);
        return (num * inv) % mod;
    }

    int countAnagrams(string s) {
        string st = "";
        long long ans = 1;
        for(int i = 0; i < s.size(); i++){
            if(s[i] == ' '){
                ans = (ans * countpermulation(st)) % mod;
                st = "";
            } else {
                st.push_back(s[i]);
            }
        }
        if(!st.empty()){
            ans = (ans * countpermulation(st)) % mod;
        }
        return ans;
    }
};



Q) codeforces Row GCD ? "https://codeforces.com/contest/1459/problem/C"
ans) in this question we find 
gj​=GCD(a1​+bj​,a2​+bj​,…,an​+bj​) but i noticed that euclid formula 
gj​=GCD(a1​+bj​,(a2​+bj​)−(a1​+bj​),(a3​+bj​)−(a1​+bj​),…) ans simplify to gj​=GCD(a1​+bj​,a2​−a1​,a3​−a1​,…,an​−a1​)

final formula is gj​=GCD(a1​+bj​,G) and G is G=GCD(∣a2​−a1​∣,∣a3​−a1​∣,…,∣an​−a1​∣)

#include <iostream>
#include <vector>
using namespace std;


int gcd(int x , int y){
if(min(x,y)==0){
    return max(x,y);
}
int mini = min(x,y);
int maxi = max(x,y);
return gcd(mini , maxi%mini);
}

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> a(n), b(m);  

    for (int i = 0; i < n; i++) cin >> a[i];
    for (int j = 0; j < m; j++) cin >> b[j];

    // Step 1: compute G = gcd of differences (relative to a[0])
    long long G = 0;
    for (int i = 1; i < n; i++) {
        G = gcd(G, abs(a[i] - a[0]));
    }

    // Step 2: answer queries
    for (int j = 0; j < m; j++) {
        cout << gcd(a[0] + b[j], G) << " ";
    }
    cout << "\n";

    return 0;
}
