Q) a^n find pow(a,n)?
ans) class Solution {
public:
    double mypowhelper(double x, long long n) {  
        if (n == 0) return 1;
        double partial = mypowhelper(x, n / 2);
        if (n % 2 == 0) {
            return partial * partial;
        } else {
            return partial * partial * x;
        }
    }

    double myPow(double x, int n) {
        long long N = n;  
        if (N < 0) {
            return 1 / mypowhelper(x, -N);  
        }
        return mypowhelper(x, N);
    }
};

Q) leetcode 344  reverse the string ?
ans)  
in this question we are swap the first and last integer until left>=right 

class Solution {
public:
    void reverseStringhelper(vector<char>& s, int left, int right) {
        if (left >= right) return;  
        swap(s[left], s[right]);  
        reverseStringhelper(s, left + 1, right - 1); 
    }

    void reverseString(vector<char>& s) {
        reverseStringhelper(s, 0, s.size() - 1);
    }
};

ans) same with above question we can check the left == right then call the function else return false 


class Solution {
  public:
    bool isPalindromehelper(string& s , int left , int right) {
        if (left >= right) return true ;  
        if(s[left]==s[right]){
            return isPalindromehelper(s,left+1 , right-1);
        }else{
            return false ; 
        }
    }
    
    bool isPalindrome(string& s) {
        return isPalindromehelper(s , 0 , s.size()-1);
    }
};

class Solution {
  public:
  
  bool isDigitSumPalindromehelper(string &n , int left ,int right) {
        if(left>=right) return true ;
        if(n[left]==n[right]){
            return isDigitSumPalindromehelper(n , left+1 , right-1);
        } else{
            return false ; 
        }
    }
    bool isDigitSumPalindrome(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;  
            n /= 10;    
        }
        string str = to_string(sum);
        return isDigitSumPalindromehelper(str , 0 , str.size()-1);
        
        
    }
};
