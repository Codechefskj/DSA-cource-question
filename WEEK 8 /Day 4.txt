     class Solution {
public:
    vector<int> mergedsortedarray(vector<int> &left, vector<int> &right) {
        int i = 0, j = 0;
        vector<int> finalans;
        while (i < left.size() && j < right.size()) {
            if (left[i] <= right[j]) {
                finalans.push_back(left[i]);
                i++;
            } else {
                finalans.push_back(right[j]);
                j++;
            }
        }
        while (i < left.size()) {
            finalans.push_back(left[i]);
            i++;
        }

        while (j < right.size()) {
            finalans.push_back(right[j]);
            j++;
        }

        return finalans;
    }

    vector<int> sortArray(vector<int>& nums) {
        int n = nums.size();
        if (n == 1) {
            return nums;
        }
        vector<int> left(nums.begin(), nums.begin() + n / 2);
        vector<int> right(nums.begin() + n / 2, nums.end());

        left = sortArray(left);
        right = sortArray(right);

        return mergedsortedarray(left, right);
    }
};

    int getCommon(vector<int>& nums1, vector<int>& nums2) {
        int i = 0, j = 0;
        while (i < nums1.size() && j < nums2.size()) {
            if (nums1[i] == nums2[j]) {
                return nums1[i];  
            }
            else if (nums1[i] < nums2[j]) {
                i++;  
            }
            else {
                j++; 
            }
        }
        return -1;
    }

    int inversionCount(vector<int> &arr) {
        int n = arr.size();
        if (n <= 1) return 0; 

        int mid = n / 2;
        vector<int> left(arr.begin(), arr.begin() + mid);
