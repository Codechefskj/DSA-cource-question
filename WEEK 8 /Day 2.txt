class Solution {
public:
    long long factorial(int num) {
        long long fact = 1;
        for (int i = 2; i <= num; i++) {
            fact *= i;
        }
        return fact;
    }

    vector<int> nthRowOfPascalTriangle(int n) {
        vector<int> finalans;

        for (int i = 0; i < n; i++) {
            long long numerator = factorial(n - 1);
            long long denominator = factorial(i) * factorial(n - 1 - i);
            finalans.push_back(numerator / denominator);
        }

        return finalans;
    }
};

Q) gfg "https://practice.geeksforgeeks.org/contest/job-a-thon-11-hiring-challenge/problems" rectangle and square?
ans) we have two input A and B . A is row ans B is coloume then we move C  that colour the hole grid 
     total grid area are A*B and we have area C*C we can select the minimum no of the move 

     solution is : 
     We need to paint a rectangular grid of size A × B using the minimum number of square tiles of any size C × C

     Approach 
     -> Start with the given dimensions (A, B).
     -> Find the largest square size possible (which is the smaller of A and B).
     -> Count how many such squares fit in the grid.
     -> Reduce the remaining unpainted area and repeat until fully painted.
     -> Return the total number of moves


     class Solution {
  public:
    int minimumMoves(int A, int B) {
        int moves = 0;                             //We initialize moves = 0 to count the number of squares used.
        while (A > 0 && B > 0) {                   // We continue the loop while both A > 0 and B > 0.
            int side = min(A, B);                  //  Choose the largest square possible
            moves += (A / side) * (B / side);      //   Count number of such squares
            if (A > B) A = A % side;               // Reduce A
            else B = B % side;                     // Reduce B
        }
        return moves;
    }
};
