Q) leetcode 744. Find Smallest Letter Greater Than Target?
ans) class Solution {
public:
    char nextGreatestLetterHelper(vector<char>& letters, char target, int l, int r, char ans) {
        if (l > r) return ans;

        int mid = l + (r - l) / 2;
        
        if (letters[mid] > target) {
            ans = letters[mid]; 
            return nextGreatestLetterHelper(letters, target, l, mid - 1, ans);
        } else {
            return nextGreatestLetterHelper(letters, target, mid + 1, r, ans);
        }
    }

    char nextGreatestLetter(vector<char>& letters, char target) {
        return nextGreatestLetterHelper(letters, target, 0, letters.size() - 1, letters[0]);
    }
};

Q) leetcode 34. Find First and Last Position of Element in Sorted Array ?
ans) class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> finalans; 
        for (int j = 0; j < nums.size(); j++) {
            if (nums[j] == target) {
                finalans.push_back(j);
            }
        }
        if (finalans.size() == 0) {
            return { -1, -1 };  
        }
        if (finalans.size() == 1) {
            finalans.push_back(finalans[0]); 
        } else {
            return {finalans[0], finalans.back()};  
        }
        return finalans;  
    }
};

Q) gfg "https://www.geeksforgeeks.org/problems/number-of-occurrence2259/1" ?
ans) class Solution {
public:
    int countFreqHelper(vector<int>& arr, int target, int l, int r) {
        if (l > r) return 0;

        int mid = l + (r - l) / 2;
        
        if (arr[mid] == target) {
            return 1 + countFreqHelper(arr, target, l, mid - 1)
                     + countFreqHelper(arr, target, mid + 1, r);
        }
        else if (target < arr[mid]) {
            return countFreqHelper(arr, target, l, mid - 1);
        } else {
            return countFreqHelper(arr, target, mid + 1, r);
        }
    }

    int countFreq(vector<int>& arr, int target) {
        return countFreqHelper(arr, target, 0, arr.size() - 1);
    }
};

Q) leetcode 704. Binary Search ?
ans) class Solution {
public:
    int searchHelper(vector<int>& nums, int target, int l, int r) {
        if (l > r) return -1; 

        int mid = (l + r) / 2;

        if (nums[mid] == target) {
            return mid;
        }

        if (nums[mid] < target) {
            return searchHelper(nums, target, mid + 1, r);
        } else {
            return searchHelper(nums, target, l, mid - 1);
        }
    }

    int search(vector<int>& nums, int target) {
        return searchHelper(nums, target, 0, nums.size() - 1);
    }
};

Q) leetcode 153. Find Minimum in Rotated Sorted Array?
ans) class Solution {
public:
    int findMin(vector<int>& nums) {
        int l = 0, r = nums.size() - 1;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (nums[mid] > nums[r]) {
                l = mid + 1;  
            } else {
                r = mid;  
            }
        }
        return nums[l]; 
    }
};
