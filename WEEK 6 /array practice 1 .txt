class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {
        int head = 0;
        int tail = nums.size() - 1;
        vector<int> result(nums.size()); 
        int pos = nums.size() - 1;     

        while (head <= tail) { 
            if (abs(nums[head]) > abs(nums[tail])) {
                result[pos] = nums[head] * nums[head];
                head++;
            } else {
                result[pos] = nums[tail] * nums[tail];
                tail--;
            }
            pos--; 
        }
        return result;
    }
};
