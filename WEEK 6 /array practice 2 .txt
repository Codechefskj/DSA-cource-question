Q) leetcode 2574 
ans) in this question i am created a suffix sum and prefix sum the answer is prefix sum - suffix sum 

class Solution {
public:
    vector<int> leftRightDifference(vector<int>& nums) {
        int n = nums.size();
        
        // Create prefix sum
        vector<int> prefix(n, 0);
        prefix[0] = nums[0];
        for (int i = 1; i < n; i++) {
            prefix[i] = prefix[i - 1] + nums[i];
        }
        
        // Create suffix sum
        vector<int> suffix(n, 0);
        suffix[n - 1] = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            suffix[i] = suffix[i + 1] + nums[i];
        }
        
        // Calculate the result
        vector<int> result(n, 0);
        for (int k = 0; k < n; k++) {
            int leftSum = (k == 0) ? 0 : prefix[k - 1];
            int rightSum = (k == n - 1) ? 0 : suffix[k + 1];
            result[k] = abs(leftSum - rightSum);
        }
        return result;
    }
};

     class Solution {
public:
    vector<long long> findPrefixScore(vector<int>& nums) {
        int n = nums.size();
        
        vector<int> maximum(n, 0);
        maximum[0] = nums[0];
        for (int i = 1; i < n; i++) {
            maximum[i] = max(maximum[i - 1], nums[i]);
        }
        
        // Calculate the final prefix score
        vector<long long> result(n,0);
        long long prefixSum = 0;
        for (int j = 0; j < n; j++) {
            prefixSum += nums[j] + maximum[j];
            result[j] = prefixSum;
        }
        
        return result;
    }
};

class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        int size = nums.size();
        // creating new vector positive or negative 
        vector <int> positive;
        vector <int> negative;
        for(int num : nums){
            if(num>0){
                positive.push_back(num);
            }else {
                negative.push_back(num);
            }
        }
        // giving the result 
        vector <int> result(size,0); 
        int positiveidx = 1 ;
        int negativeidx = 0 ;
        result[0] = positive[0];
        for(int i = 1 ; i<size ; i++){
        if(i%2==0){
            result[i]= positive[positiveidx++];
        }else{
            result[i]=negative[negativeidx++];
        }
        } return result ; 
    }
};

class Solution {
public:
    int alternatingSubarray(vector<int>& nums) {
        int maxLength = -1; 
        int currentLength = 1; 

        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] == nums[i - 1] + 1 && (currentLength % 2 == 1) ||
                nums[i] == nums[i - 1] - 1 && (currentLength % 2 == 0)) {
                currentLength++;
            } else if (nums[i] == nums[i - 1] + 1) {
                currentLength = 2;
            } else {
                currentLength = 1;
            }
            maxLength = max(maxLength, currentLength);
        }
        return maxLength >= 2 ? maxLength : -1; 
    }
};
